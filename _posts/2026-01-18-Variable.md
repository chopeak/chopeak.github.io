```c
/* swap.c */
/* 交换a, b */

#include <stdio.h>

int main(void)
{
    int a, b;
    a = 5;
    b = 6;
    
    int t;
    t = a;
    a = b;
    b = t;
    
    printf("a = %d; b = %d\n", a, b);
    return 0;
}
```
